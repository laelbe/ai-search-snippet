</search-bar-snippet>

    placeholder="Search documentation..."

</search-modal-snippet>

<!-- 4. Full Page Chat (use in dedicated page) -->

App.tsx 
import { useEffect, useRef } from 'react';
import '@anthropic/search-snippet';

// Wrapper component for type-safe props
function SearchBar({ apiUrl, placeholder, maxResults }: {
  apiUrl: string;
  placeholder?: string;
  maxResults?: number;
}) {
  const ref = useRef<HTMLElement>(null);

  useEffect(() => {
    // Web Components work directly in React 19+
    // For older versions, set attributes manually
    if (ref.current) {
      ref.current.setAttribute('api-url', apiUrl);
      if (placeholder) ref.current.setAttribute('placeholder', placeholder);
      if (maxResults) ref.current.setAttribute('max-results', String(maxResults));
    }
  }, [apiUrl, placeholder, maxResults]);

  return <search-bar-snippet ref={ref} />;
}

// Usage in your app
export default function App() {
  return (
    <div>
      {/* Search Bar */}
      <SearchBar 
        apiUrl="https://api.example.com"
        placeholder="Search..."
        maxResults={10}
      />

      {/* Chat Bubble - works directly as Web Component */}
      <chat-bubble-snippet 
        api-url="https://api.example.com"
        style={{ '--search-snippet-primary-color': '#F6821F' } as React.CSSProperties}
      />

      {/* Search Modal with keyboard shortcut */}
      <search-modal-snippet 
        api-url="https://api.example.com"
        placeholder="Search documentation..."
        shortcut="k"
      />
    </div>
  );
}

// TypeScript declarations (add to global.d.ts)

}
App.vue 
<script setup lang="ts">
import '@anthropic/search-snippet';
import { ref } from 'vue';

const apiUrl = 'https://api.example.com';
const searchPlaceholder = ref('Search...');
const maxResults = ref(10);

// Handle custom events from components
function onSearchComplete(event: CustomEvent) {
  console.log('Search results:', event.detail);
}

function onChatMessage(event: CustomEvent) {
  console.log('Chat message:', event.detail);
}
</script>

<template>
  <div>
    <!-- Search Bar with v-bind -->
    <search-bar-snippet
      :api-url="apiUrl"
      :placeholder="searchPlaceholder"
      :max-results="maxResults"
      @search-complete="onSearchComplete"
    />

    <!-- Chat Bubble Widget -->
    <chat-bubble-snippet
      :api-url="apiUrl"
      placeholder="Ask me anything..."
      @chat-message="onChatMessage"
    />

    <!-- Search Modal (Cmd/Ctrl+K) -->
    <search-modal-snippet
      :api-url="apiUrl"
      placeholder="Search documentation..."
      shortcut="k"
    />

    <!-- Full Page Chat -->
    <chat-page-snippet
      :api-url="apiUrl"
      class="chat-container"
    />
  </div>
</template>

<style>
/* Customize with CSS Variables */
search-bar-snippet,
chat-bubble-snippet {
  --search-snippet-primary-color: #F6821F;
  --search-snippet-border-radius: 12px;
}

.chat-container {
  height: 500px;
}
</style>
