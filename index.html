Quick Start Guide

Add search and chat to your website in minutes.

Installation & Usage

<!-- Import the library -->
<script type="module" src="https://cdn.example.com/search-snippet.es.js"></script>

<!-- 1. Chat Bubble Widget (Fixed bottom-right) -->
<chat-bubble-snippet 
    api-url="https://api.example.com"
</chat-bubble-snippet>

<!-- 2. Search Bar Widget -->
<search-bar-snippet 
    api-url="https://api.example.com"
    placeholder="Search..."
    max-results="10">
</search-bar-snippet>

<!-- 3. Search Modal (Cmd/Ctrl+K to open) -->
<search-modal-snippet 
    api-url="https://api.example.com"
    placeholder="Search documentation..."
    shortcut="k">
</search-modal-snippet>

<!-- 4. Full Page Chat (use in dedicated page) -->
<chat-page-snippet 
    api-url="https://api.example.com"
</chat-page-snippet>

<!-- Customize with CSS Variables -->
<style>
    chat-bubble-snippet {
        --search-snippet-primary-color: #F6821F;
        --search-snippet-border-radius: 12px;
    }
</style>
