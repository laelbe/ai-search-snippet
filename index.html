📖 Quick Start

<!-- Import the library -->
<script type="module" src="https://cdn.example.com/search-snippet.es.js"></script>

<!-- 1. Chat Bubble Widget (Fixed bottom-right) -->
<chat-bubble-snippet 
    api-url="https://api.example.com"
    enable-streaming="true">
</chat-bubble-snippet>

<!-- 2. Search Bar Widget -->
<search-bar-snippet 
    api-url="https://api.example.com"
    placeholder="Search..."
    max-results="10">
</search-bar-snippet>

<!-- 3. Full Page Chat (use in dedicated page) -->
<chat-page-snippet 
    api-url="https://api.example.com"
    enable-streaming="true">
</chat-page-snippet>

<!-- Customize with CSS -->
<style>
    chat-bubble-snippet {
        --search-snippet-primary-color: #667eea;
        --search-snippet-border-radius: 12px;
    }
</style>
